func TestDay0Part1(t *testing.T) {
	RegisterTestingT(t)

	start := time.Now()
	result, err := aoc2025.Day0Part1(strings.NewReader(day0ExampleInput))
	end := time.Now()

	Expect(err).NotTo(HaveOccurred())

}
